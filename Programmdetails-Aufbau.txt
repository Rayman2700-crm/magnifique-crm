Was ist kostenlos (realistisch) und wo sind die Grenzen?
Hosting der Web-App (Next.js)

Vercel Hobby ist gratis und der Standard für Next.js Deployments.

Für ein Studio-Tool mit wenigen Nutzern (4–5 Behandler + Admin) reicht das in der Regel locker.

Achtung: Wenn ihr sehr viele Bilder über Vercel ausliefert (oder extrem viel Traffic habt), können Limits/„Fair Use“ irgendwann relevant werden. Die Details hängen stark vom Nutzungsverhalten ab.

Backend (DB, Auth, Storage) – Supabase

Supabase hat einen Free Plan, der für MVP/Start super ist. Die wichtigsten Größen, die du im Blick behalten musst:

DB Größe: 500 MB pro Projekt (Free)

Egress (Traffic raus): 5 GB (Free)

Auth (MAU): bis 50.000 monatlich aktive User (Free)

Für euch: 4–5 interne Nutzer + Kunden füllen Fragebogen im Studio (kein öffentlicher Login) → MAU ist völlig unkritisch. Kritischer wird eher Storage/Egress, wenn ihr sehr viele Fotos/PDFs speichert und oft herunterladet.

Besser für Solo-Entwicklung + schnell live: Next.js + Supabase (ja).

Am längsten kostenlos nutzbar: auch diese Kombi, wenn wir Bilder/PDFs effizient behandeln (Komprimierung, Thumbnails, keine unnötigen Downloads).

import Link from "next/link";
import { supabaseServer } from "@/lib/supabase/server";
import { setAdminTenant } from "@/app/admin/actions";
import { getAdminTenantCookie, getEffectiveTenantId } from "@/lib/effectiveTenant";
import AdminTenantSelect from "./AdminTenantSelect";

export default async function CustomersPage({
  searchParams,
}: {
  searchParams?: { q?: string } | Promise<{ q?: string }>;
}) {
  const sp = searchParams ? await searchParams : undefined;
  const qRaw = (sp?.q ?? "").toString();
  const q = qRaw.trim().toLowerCase();

  const supabase = await supabaseServer();
  const { data: userData } = await supabase.auth.getUser();
  const user = userData.user;

  if (!user) {
    return (
      <main className="p-6">
        <Link href="/login" className="underline">
          Bitte einloggen
        </Link>
      </main>
    );
  }

  const { data: profile } = await supabase
    .from("user_profiles")
    .select("role, tenant_id, full_name")
    .eq("user_id", user.id)
    .single();

  const role = (profile?.role ?? "PRACTITIONER") as string;

  // Admin Dropdown-Optionen
  let tenantOptions: { tenant_id: string; label: string }[] = [];
  let currentAdminTenant = "all";

  if (role === "ADMIN") {
    currentAdminTenant = await getAdminTenantCookie();

    const { data: practitioners } = await supabase
      .from("user_profiles")
      .select("tenant_id, full_name")
      .eq("role", "PRACTITIONER");

    const seen = new Set<string>();
    tenantOptions =
      (practitioners ?? [])
        .filter((p: any) => p.tenant_id)
        .map((p: any) => ({
          tenant_id: p.tenant_id as string,
          label: (p.full_name as string) || (p.tenant_id as string),
        }))
        .filter((x) => {
          if (seen.has(x.tenant_id)) return false;
          seen.add(x.tenant_id);
          return true;
        })
        .sort((a, b) => a.label.localeCompare(b.label));
  }

  const effectiveTenantId = await getEffectiveTenantId({
    role: profile?.role ?? "PRACTITIONER",
    tenant_id: profile?.tenant_id ?? null,
  });

  // Kunden-Query
  let queryBuilder = supabase
    .from("customer_profiles")
    .select(
      `
        id,
        created_at,
        tenant_id,
        person_id,
        person:persons (
          id,
          full_name,
          phone,
          email,
          birthday
        )
      `
    )
    .order("created_at", { ascending: false })
    .limit(200);

  if (role === "ADMIN" && effectiveTenantId) {
    queryBuilder = queryBuilder.eq("tenant_id", effectiveTenantId);
  }

  const { data: rowsRaw, error } = await queryBuilder;

  const rows = (rowsRaw ?? []).filter((r: any) => {
    if (!q) return true;

    const name = (r?.person?.full_name ?? "").toString().toLowerCase();
    const phone = (r?.person?.phone ?? "").toString().toLowerCase();
    const email = (r?.person?.email ?? "").toString().toLowerCase();

    return name.includes(q) || phone.includes(q) || email.includes(q);
  });

  const adminMissingOwnTenant =
    role === "ADMIN" && currentAdminTenant === "all" && !profile?.tenant_id;

  return (
    <main className="p-6 mx-auto max-w-5xl">
      <div className="flex items-start justify-between gap-6">
        <div>
          <h1 className="text-2xl font-semibold text-white">Kunden</h1>
          <p className="mt-1 text-sm text-white/60">
            Eingeloggt als {profile?.full_name ?? user.email} ({profile?.role ?? "—"})
          </p>

          {role === "ADMIN" ? (
            <div className="mt-3 space-y-2">
              <AdminTenantSelect current={currentAdminTenant} options={tenantOptions} action={setAdminTenant} />

              {currentAdminTenant === "all" ? (
                <div className="text-xs text-white/50">
                  Hinweis: “Alle” ist die Admin-Ansicht. “Neuer Kunde” wird dann in deinem eigenen Tenant angelegt.
                </div>
              ) : null}

              {adminMissingOwnTenant ? (
                <div className="text-xs text-red-300">
                  Dein ADMIN-Profil hat aktuell kein tenant_id. Bitte setze tenant_id in user_profiles, sonst kann “Alle → eigener Tenant”
                  nicht funktionieren.
                </div>
              ) : null}
            </div>
          ) : null}
        </div>

        <Link
          className="rounded-xl bg-white text-black px-4 py-2 font-medium hover:bg-white/90"
          href="/customers/new"
        >
          + Neuer Kunde
        </Link>
      </div>

      {/* Suche */}
      <div className="mt-6 flex items-center gap-3">
        <form className="flex w-full max-w-md items-center gap-2" action="/customers" method="get">
          <input
            name="q"
            defaultValue={qRaw}
            placeholder="Suchen: Name, Telefon oder E-Mail…"
            className="w-full rounded-xl border border-white/10 bg-black/30 px-3 py-2 text-sm text-white placeholder:text-white/40 outline-none focus:border-white/20"
          />
          <button
            type="submit"
            className="rounded-xl border border-white/10 bg-white/10 px-4 py-2 text-sm text-white hover:bg-white/15"
          >
            Suchen
          </button>

          {qRaw ? (
            <Link
              href="/customers"
              className="rounded-xl border border-white/10 bg-black/20 px-3 py-2 text-sm text-white/80 hover:bg-black/30"
            >
              Reset
            </Link>
          ) : null}
        </form>

        <div className="ml-auto text-sm text-white/50">
          {rows.length} Ergebnis(se)
        </div>
      </div>

      {error ? (
        <div className="mt-4 rounded-xl border border-red-500/30 bg-red-500/10 p-3 text-sm text-red-200">
          Fehler: {error.message}
        </div>
      ) : null}

      <div className="mt-6 overflow-hidden rounded-2xl border border-white/10 bg-[var(--surface)]">
        <table className="w-full text-sm">
          <thead className="bg-white/5">
            <tr className="text-left text-white/70">
              <th className="p-3 font-semibold">Name</th>
              <th className="p-3 font-semibold">Telefon</th>
              <th className="p-3 font-semibold">E-Mail</th>
              <th className="p-3 font-semibold">Erstellt</th>
            </tr>
          </thead>

          <tbody>
            {rows.map((r: any) => {
              const href = `/customers/${r.id}`;
              return (
                <tr key={r.id} className="border-t border-white/10 hover:bg-white/5">
                  <td className="p-3 font-medium text-white">
                    <Link className="block" href={href}>
                      {r.person?.full_name ?? "-"}
                    </Link>
                  </td>

                  <td className="p-3 text-white/80">
                    <Link className="block" href={href}>
                      {r.person?.phone ?? "-"}
                    </Link>
                  </td>

                  <td className="p-3 text-white/80">
                    <Link className="block" href={href}>
                      {r.person?.email ?? "-"}
                    </Link>
                  </td>

                  <td className="p-3 text-white/70">
                    <Link className="block" href={href}>
                      {r.created_at ? new Date(r.created_at).toLocaleDateString() : "-"}
                    </Link>
                  </td>
                </tr>
              );
            })}

            {(rows?.length ?? 0) === 0 ? (
              <tr>
                <td className="p-6 text-white/50" colSpan={4}>
                  Noch keine Kunden vorhanden.
                </td>
              </tr>
            ) : null}
          </tbody>
        </table>
      </div>
    </main>
  );
}
